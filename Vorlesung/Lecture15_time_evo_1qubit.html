def Exact_evol(ini_state, t, hh, G):

    Ut = QuantumCircuit(1)
    
    eta = np.sqrt( hh*hh + G*G )
    st2 = (G/eta) * np.sin(eta*t/2.)
    theta2 = np.arcsin(st2)
    tan2t1 = - eta /(hh * np.tan(eta*t/2.) )
    theta1 = np.arctan(tan2t1) /2.
    theta3 = theta1 + pi/2.

    Ut.rz(- 2. * theta1,0)
    Ut.ry(- 2. * theta2,0)
    Ut.rz(- 2. * theta3,0)

    evolved_state = ini_state.compose(Ut)

    return evolved_state

def energy_longit(evolved_state, t, hh, G, sim, n_shots):

    qc = QuantumCircuit(1,1)
    qc = evolved_state.compose(qc)
    qc.measure(0,0)

    #display(qc.draw('mpl'))

    qc_trans = transpile(qc, sim)
    counts = sim.run(qc_trans, shots=n_shots).result().get_counts()

    # get the probabilities
    probs = {}
    for output in ['0','1']:
        if output in counts:
            probs[output] = counts[output]/n_shots
        else:
            probs[output] = 0

    E_long = probs['0'] -  probs['1'] 
    E_long = E_long * (-hh/2.)

    return E_long

def energy_transv(evolved_state, t, hh, G, sim, n_shots):

    qc = QuantumCircuit(1,1)
    qc = evolved_state.compose(qc)
    qc.h(0)
    qc.measure(0,0)

    #display(qc.draw('mpl'))

    qc_trans = transpile(qc, sim)
    counts = sim.run(qc_trans, shots=n_shots).result().get_counts()

    probs = {}
    for output in ['0','1']:
        if output in counts:
            probs[output] = counts[output]/n_shots
        else:
            probs[output] = 0

    E_transv = probs['0'] -  probs['1'] 
    E_transv = E_transv * (-G/2.)

    return E_transv

#### Trotterization to leading order
# circuit building |Psi(t)> = U(t)_LO |Psi(0)>

def Trotter_evol(ini_state, t, Ns, hh, G):

    dt = t/Ns
    time = 0.

    Ut = QuantumCircuit(1)

    # implement (e^{-iH Delta_t})^N_s to leading order
    for it in range(Ns):
        Ut.rz(-hh * dt, 0)
        Ut.rx(-G * dt, 0)
        
    evolved_state = ini_state.compose(Ut)
        
    return evolved_state

def Trotter_evol_ord2(ini_state, t, Ns, hh, G):

    dt = t/Ns
    time = 0.

    Ut = QuantumCircuit(1)

    # implement (e^{-iH Delta_t})^N_s to leading order
    for it in range(Ns):
        Ut.rx(-G * dt, 0)
        Ut.rz(-hh * dt, 0)
        
    evolved_state = ini_state.compose(Ut)
        
    return evolved_state

def Trotter_evol_corr(ini_state, t, Ns, hh, G):

    dt = t/Ns
    time = 0.
    #print("dt",dt)

    Ut = QuantumCircuit(1)
    
    for it in range(Ns):
        Ut.ry(hh * G * dt * dt /2. , 0)
        Ut.rz(-hh * dt, 0)
        Ut.rx(-G * dt, 0)
        
    evolved_state = ini_state.compose(Ut)

    #display(evolved_state.draw('mpl'))
        

    return evolved_state

def Trotter_evol_NLO_ord1(ini_state, t, Ns, hh, G):

    dt = t/Ns
    time = 0.

    Ut = QuantumCircuit(1)

    for it in range(Ns):

        Ut.rz(-hh * dt, 0)
        Ut.rx(-G * dt/2, 0)
        
    evolved_state = ini_state.compose(Ut)
